'🔐' : '🔓', username: data.username, id: data.id, message: '✅ توكن شغال بالكامل' }; } // الحالة 4: غير موثق (كلاهما غير موثق) if (!emailVerified && !phoneVerified) { return { status: 'unverified', email: '🔒', phone: '🔒', nitro: '❌', mfa: '❌', username: data.username, id: data.id, message: '⚠️ حساب غير موثق - بريد وهاتف غير موثقين' };
